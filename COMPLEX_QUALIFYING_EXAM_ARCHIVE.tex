\item[id=misc, id=S06,tag=S06.3.]
The Bernoulli polynomials $\varphi_n(z)$ are defined by the expansion
\[
	\frac{e^{tz} - 1}{e^t -1} = \sum_{n=1}^{\infty} \frac{\varphi_n(z)}{n!} t^{n-1}
\]
Prove the following two statements
\begin{description}\onlyitems
	\item[(i)] $\varphi_n(z+1) - \varphi_n(z) = nz^{n-1}$
	\item[(ii)] $\frac{\varphi_{n+1}(n+1)}{n+1} = 1 + 2^n + 3^n + \ldots + n^n$ 
\end{description}

	h(x) = \int_{-\infty}^{\infty} f(x+iy) dy
\]
is a constant function of $x \in [\alpha_1, \alpha_2]$.
